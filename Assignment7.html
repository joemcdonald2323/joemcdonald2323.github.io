  1. Make a plot using position ='jitter'.
df %>% filter(state=='NJ', dataQualityGrade=='A+') %>% ggplot()+

  1. Make a plot using position ='dodge'.
df %>% filter(state=='FL') %>% ggplot()+
  geom_bar(mapping = aes(x=dataQualityGrade),
           position='dodge')

  1. Make a plot using position ='fill'.
df %>% filter(state=='FL') %>% ggplot()+
  geom_bar(mapping = aes(x=dataQualityGrade),
           position = 'fill')

  1. Make a plot using geom_col.
df %>% group_by(dataQualityGrade) %>% summarise(median=median(negativeIncrease)) %>% ggplot()+
  geom_col(mapping = aes(x=dataQualityGrade, y=median))
